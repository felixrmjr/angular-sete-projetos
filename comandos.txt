--------------------------------------------
Instalar o Angular CLI de Forma Global
--------------------------------------------
npm install -g @angular/cli

--------------------------------------------
Verificar versão do Angular CLI
--------------------------------------------
ng --version

--------------------------------------------
Criar um novo projeto
--------------------------------------------
ng new ola-mundo

--------------------------------------------
Subir o servidor
--------------------------------------------
ng serve

--------------------------------------------
Executar testes unitários
--------------------------------------------
ng teste

--------------------------------------------
Executar testes de integração
--------------------------------------------
ng e2e

--------------------------------------------
Verificar se o codigo está padronizado
--------------------------------------------
ng lint

--------------------------------------------
Gera o build de produção
--------------------------------------------
ng build --prod

--------------------------------------------
Gera um novo modulo
--------------------------------------------
// (g = atalho de gerador) - (module = modulo)
ng g module nome

--------------------------------------------
Gera um novo componente
--------------------------------------------
// (g = atalho de gerador) - (component = compomente)
ng g component nome

--------------------------------------------
Gera um novo serviço
--------------------------------------------
// (g = atalho de gerador) - (service = serviço)
ng g service nome
